## 1. 技术栈选型

- 统一语言：全栈 TypeScript
- Monorepo管理：pnpm workspace + Turborepo
- 最新版本优先（紧跟上游最新发布，非仅LTS）
- 性能优先：SSR/SSG、边缘缓存、懒加载

### 2.2 请求流程（作文批改）

```
1. 学生提交作文
   └── POST /api/essays (写入DB, 状态: pending)

2. 入队批改任务
   └── BullMQ.add('essay:correct', { essayId })

3. AI Worker 消费队列
   ├── 读取作文内容 + 题目要求
   ├── 调用 AI API (DeepSeek/GPT-4o)
   │   ├── Step 1: 内容分析 (要点覆盖、切题度)
   │   ├── Step 2: 语言分析 (语法错误、词汇、句式)
   │   ├── Step 3: 结构分析 (段落、衔接、格式)
   │   └── Step 4: 综合评分 (按深圳中考标准)
   └── 写入批改结果到DB

4. 前端轮询/WebSocket 获取结果
   └── 状态变为 completed
```

### 2.3 多AI Provider路由策略

```
路由策略:
├── 默认路由: DeepSeek-V3 (性价比高，中文支持好)
├── 高精度路由: GPT-4o / Claude 3.5 Sonnet (复杂批改)
├── 备用路由: 通义千问 / 豆包 (国内备用)
└── 降级策略: 主模型超时→备用模型→返回部分结果
```

### 4.2 核心表 Schema

```sql
-- 学校表
CREATE TABLE schools (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(255) NOT NULL,
  region VARCHAR(50) NOT NULL,          -- 福田/南山/罗湖/宝安/龙岗/龙华/光明/坪山/盐田/大鹏
  contact_name VARCHAR(100),
  contact_phone VARCHAR(20),
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 用户表
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email VARCHAR(255) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  name VARCHAR(100) NOT NULL,
  role VARCHAR(20) NOT NULL CHECK (role IN ('super_admin', 'school_admin', 'teacher', 'student')),
  school_id UUID REFERENCES schools(id),
  student_no VARCHAR(50),               -- 学号（学生专用）
  avatar_url TEXT,
  is_active BOOLEAN DEFAULT true,
  last_login_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 班级表
CREATE TABLE classes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  school_id UUID NOT NULL REFERENCES schools(id),
  name VARCHAR(100) NOT NULL,           -- 如"初三(1)班"
  grade VARCHAR(20) NOT NULL,           -- 初一/初二/初三
  teacher_id UUID REFERENCES users(id), -- 班主任
  academic_year VARCHAR(9),             -- 如"2025-2026"
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 班级成员表（学生-班级关系）
CREATE TABLE class_enrollments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  class_id UUID NOT NULL REFERENCES classes(id) ON DELETE CASCADE,
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  role VARCHAR(20) NOT NULL DEFAULT 'student' CHECK (role IN ('student', 'teacher')),
  UNIQUE(class_id, user_id)
);

-- 作文任务表
CREATE TABLE essay_tasks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  class_id UUID NOT NULL REFERENCES classes(id),
  created_by UUID NOT NULL REFERENCES users(id),
  title VARCHAR(255) NOT NULL,
  topic_type VARCHAR(30) NOT NULL,      -- letter/speech/argumentation/narration/proposal
  topic_category VARCHAR(50),           -- school_life/social_issues/culture/tech/growth/travel
  requirements TEXT NOT NULL,           -- 题目要求
  key_points JSONB,                     -- 评分要点 ["要点1", "要点2", "要点3"]
  reference_essay TEXT,                 -- 参考范文
  word_limit_min INT DEFAULT 80,
  word_limit_max INT DEFAULT 125,
  time_limit_minutes INT DEFAULT 15,
  status VARCHAR(20) DEFAULT 'draft' CHECK (status IN ('draft', 'published', 'closed')),
  due_date TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 作文表
CREATE TABLE essays (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  task_id UUID REFERENCES essay_tasks(id),
  student_id UUID NOT NULL REFERENCES users(id),
  title VARCHAR(255),
  content TEXT NOT NULL,
  word_count INT NOT NULL,
  submit_type VARCHAR(20) DEFAULT 'typed' CHECK (submit_type IN ('typed', 'ocr')),
  ocr_image_url TEXT,                   -- OCR原始图片URL
  ocr_confidence DECIMAL(3,2),          -- OCR识别置信度 0.00-1.00
  
  -- 书写质量评估（OCR提交时）
  handwriting_score DECIMAL(3,1),       -- 书写质量评分 0-1.5
  handwriting_details JSONB DEFAULT '{}', -- {neatness, letterForm, alignment, correctionsCount}
  
  status VARCHAR(20) DEFAULT 'pending' CHECK (status IN ('pending', 'correcting', 'completed', 'failed')),
  total_score DECIMAL(4,1),
  score_tier VARCHAR(10),               -- 1st/2nd/3rd/4th/5th
  correction_id UUID,                   -- 批改结果关联
  teacher_review TEXT,                   -- 教师评语
  teacher_score DECIMAL(4,1),           -- 教师调整后的分数
  submitted_at TIMESTAMPTZ DEFAULT NOW(),
  corrected_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 批改结果表
CREATE TABLE corrections (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  essay_id UUID NOT NULL REFERENCES essays(id) ON DELETE CASCADE,
  
  -- 分维度得分
  content_score DECIMAL(3,1),
  language_score DECIMAL(3,1),
  structure_score DECIMAL(3,1),
  presentation_score DECIMAL(3,1),
  total_score DECIMAL(4,1),
  score_tier VARCHAR(10),
  
  -- 错误详情 (JSON)
  errors JSONB DEFAULT '[]',            -- [{type, original, corrected, explanation, position}]
  error_stats JSONB DEFAULT '{}',       -- {tense: 3, spelling: 1, ...}
  
  -- 亮点
  highlights JSONB DEFAULT '[]',        -- [{sentence, type, comment}]
  
  -- 逐句分析
  sentence_analysis JSONB DEFAULT '[]', -- [{original, comment, is_highlight}]
  
  -- 修改版
  revised_essay TEXT,
  
  -- 改进建议
  suggestions JSONB DEFAULT '[]',       -- [{priority, category, suggestion}]
  
  -- 元数据
  ai_provider VARCHAR(50),              -- 使用的AI模型
  ai_model VARCHAR(50),
  correction_time_ms INT,               -- 批改耗时(ms)
  
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 错题本
CREATE TABLE error_books (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES users(id),
  error_type VARCHAR(30) NOT NULL,      -- tense/spelling/agreement/preposition/...
  original_text TEXT NOT NULL,
  corrected_text TEXT NOT NULL,
  essay_id UUID REFERENCES essays(id),
  occurrence_count INT DEFAULT 1,
  last_seen_at TIMESTAMPTZ DEFAULT NOW(),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, error_type, original_text)
);

-- AI API 配置表
CREATE TABLE api_configs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  provider VARCHAR(30) NOT NULL,        -- openai/deepseek/claude/qwen/doubao
  api_key_encrypted TEXT NOT NULL,
  base_url TEXT,
  model VARCHAR(50),
  is_active BOOLEAN DEFAULT true,
  priority INT DEFAULT 0,               -- 优先级，数字越小越优先
  max_tokens INT DEFAULT 4096,
  temperature DECIMAL(2,1) DEFAULT 0.3,
  rate_limit_per_min INT DEFAULT 60,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- API 调用日志
CREATE TABLE api_call_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  provider VARCHAR(30) NOT NULL,
  model VARCHAR(50),
  endpoint VARCHAR(255),
  tokens_used INT,
  latency_ms INT,
  cost DECIMAL(10,6),
  status VARCHAR(20),                   -- success/error/timeout
  error_message TEXT,
  essay_id UUID REFERENCES essays(id),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 成就表
CREATE TABLE achievements (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES users(id),
  achievement_type VARCHAR(50) NOT NULL, -- grammar_master/full_score/improvement_king/...
  title VARCHAR(100) NOT NULL,
  description TEXT,
  icon VARCHAR(50),
  unlocked_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, achievement_type)
);

-- 系统公告
CREATE TABLE announcements (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title VARCHAR(255) NOT NULL,
  content TEXT NOT NULL,
  target_role VARCHAR(20),              -- all/super_admin/school_admin/teacher/student
  is_active BOOLEAN DEFAULT true,
  created_by UUID REFERENCES users(id),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
```
