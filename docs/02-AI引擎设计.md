```typescript
  grammarError: 0.5,        // 每个语法错误扣0.5分
  missingKeyPoint: 1.0,     // 每个要点遗漏扣1分
  formatError: 1.0,         // 格式错误扣1分
  wordCountBelow80: '降一档', // 词数不足80降一档
```

### 3.2 评分计算公式

```typescript
function calculateFinalScore(
  contentScore: number,    // 0-4.5
  languageScore: number,   // 0-6.0
  structureScore: number,  // 0-3.0
  presentationScore: number, // 0-1.5
  handwritingScore: number | null, // OCR书写质量 0-1.5 (null=typed)
  errorCount: number,
  missingPoints: number,
  hasFormatError: boolean,
  wordCount: number,
): number {
  // 基础分 = 各维度得分之和
  // 如果是OCR提交，卷面分 = AI书写质量评估分；否则默认满分
  const effectivePresentation = handwritingScore ?? presentationScore;
  let baseScore = contentScore + languageScore + structureScore + effectivePresentation;
  
  // 扣分项
  const deduction = 
    errorCount * 0.5 +           // 语法错误扣分
    missingPoints * 1.0 +        // 要点遗漏扣分
    (hasFormatError ? 1.0 : 0);  // 格式错误扣分
  
  let finalScore = baseScore - deduction;
  
  // 词数不足80，降一档（约-3分）
  if (wordCount < 80) {
    finalScore = Math.min(finalScore, 9.5); // 最多第三档
  }
  
  // 词数在100-125最佳区间，且内容质量好，可加0.5分
  if (wordCount >= 100 && wordCount <= 125 && contentScore >= 3.5) {
    finalScore = Math.min(15, finalScore + 0.5);
  }
  
  // 确保分数在0-15范围内
  return Math.max(0, Math.min(15, Math.round(finalScore * 2) / 2));
}
```

### 4.2 路由策略

```typescript
// packages/ai/router.ts
class AIProviderRouter {
  private providers: AIProvider[] = [];
  private fallbackChain: AIProvider[][] = [];
  
  /**
   * 路由策略:
   * 1. 默认: DeepSeek-V3 (性价比高)
   * 2. 高精度场景: GPT-4o (内容分析、综合评分)
   * 3. 备用: Claude 3.5 Sonnet
   * 4. 国内备用: 通义千问
   */
  async route(correctionType: 'content' | 'language' | 'structure' | 'scorer'): Promise<AIProvider> {
    switch (correctionType) {
      case 'content':
        return this.getProvider('openai');   // 内容分析用GPT-4o
      case 'language':
        return this.getProvider('deepseek');  // 语言分析用DeepSeek
      case 'structure':
        return this.getProvider('deepseek');  // 结构分析用DeepSeek
      case 'scorer':
        return this.getProvider('openai');    // 综合评分用GPT-4o
    }
  }
  
  async executeWithFallback<T>(
    correctionType: string,
    fn: (provider: AIProvider) => Promise<T>,
  ): Promise<T> {
    const primaryProvider = await this.route(correctionType);
    try {
      return await fn(primaryProvider);
    } catch (error) {
      logger.warn(`Primary provider failed, trying fallback`, { error });
      const fallbackProvider = this.getFallback();
      return await fn(fallbackProvider);
    }
  }
}
```

## 5. 错误处理与降级

### 5.1 降级策略

```
Level 1: 正常模式
├── 三个维度并行批改
├── 综合评分
└── 完整报告生成

Level 2: 降级模式（主模型超时）
├── 切换到备用模型
└── 功能不变

Level 3: 部分降级（全部模型不可用）
├── 仅返回语言维度分析（基于规则引擎）
├── 拼写检查（本地词典）
└── 不返回综合评分

Level 4: 最低降级
├── 仅返回词数统计
└── 提示用户稍后重试
```

### 6.3 书写质量计算

```typescript
function calculateHandwritingScore(details: HandwritingDetails): number {
  const { neatnessScore, letterFormScore, alignmentScore, correctionMarks } = details;
  
  // 三个维度取平均
  let score = (neatnessScore + letterFormScore + alignmentScore) / 3;
  
  // 涂改扣分：超过3处每处扣0.1分
  if (correctionMarks > 3) {
    score -= (correctionMarks - 3) * 0.1;
  }
  
  // 最终得分范围 0-1.5
  return Math.max(0, Math.min(1.5, Math.round(score * 10) / 10));
}
```
