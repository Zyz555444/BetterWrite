```typescript
async function correctEssay(essay: string, task: EssayTask): Promise<CorrectionResult> {
  const preprocessed = preprocess(essay);
  
  // 四个维度并行分析
  const [topicAdherenceResult, contentResult, languageResult, structureResult] = await Promise.all([
    analyzeTopicAdherence(essay, task, preprocessed),
    analyzeContent(essay, task, preprocessed),
    analyzeLanguage(essay, preprocessed),
    analyzeStructure(essay, task, preprocessed),

  // 综合评分
  const scoreResult = await calculateScore(
    topicAdherenceResult, contentResult, languageResult, structureResult, task
  );

  return {
    ...topicAdherenceResult,
    ...contentResult,
    ...languageResult,
    ...structureResult,
    ...scoreResult,
    aiProvider: getCurrentProvider(),
    aiModel: getCurrentModel(),

}
```

### 2.5 综合评分 Prompt

```
你是一位深圳中考英语作文阅卷老师。请根据以下各维度分析结果，进行综合评分。

## 评分原则（定档制）
真实阅卷采用“先定档、后微调”：
1. 先通读全文，根据“内容要点是否齐全、语言是否基本正确、结构是否清晰”确定所属档次。
2. 再在各维度得分上微调，但维度分必须与档次一致：缺要点不能给第一档；语言错误过多不能给第一档；结构混乱不能给第一档。
3. 内容要点是入档门槛：漏写一个核心要点，内容分直接扣 1.5-2.0 分，并可能降一档。

## 深圳中考英语作文评分标准（15分制）

| 档次 | 分数 | 标准 |
|------|------|------|
| 第一档（优） | 15-13 | 要点齐全且展开充分，语言正确无误，行文流畅，结构清晰 |
| 第二档（良） | 12-10 | 要点较完整，语言有少量错误，意思较连贯，结构基本清晰 |
| 第三档（中） | 9-7 | 要点部分缺失或展开不足，有一些语言错误，连贯性一般 |
| 第四档（及格） | 6-4 | 要点缺失较多，语言错误较多，意思表达部分清楚 |
| 第五档（差） | 3-0 | 只能写出少量相关词，不能表达完整意思 |

## 各维度满分
- 审题扣题维度: 2.0 分（文体、格式、人称、时态）
- 内容维度: 5.0 分（要点覆盖、要点展开、切题度）
- 语言维度: 4.0 分（语法正确性、词汇丰富性、句式多样性）
- 结构维度: 2.5 分（段落结构、连接词、逻辑连贯）
- 卷面维度: 1.5 分（字数达标、格式规范；打字输入默认满分，OCR 按书写质量调整）

## 各维度分析结果
- 审题扣题得分: {topicAdherenceScore}/2.0
- 内容得分: {contentScore}/5.0
- 语言得分: {languageScore}/4.0
- 结构得分: {structureScore}/2.5

## 要点覆盖情况
{pointCoverage}

## 审题扣题问题
{topicAdherenceIssues}

## 语言错误
{languageErrors}

## 词数
当前词数: {wordCount}
要求范围: {wordLimitMin}-{wordLimitMax}词

## 扣分与定档规则
- 每遗漏一个核心要点：内容分扣 1.5-2.0 分，并可能导致降档
- 每个关键语法/时态/主谓一致错误：语言分扣 0.5-1.0 分
- 格式错误（如书信无称呼/结尾）：审题扣题或结构分扣 0.5-1.5 分
- 字数不足下限：最高只能进入第三档（≤9分）
- 字数在最佳区间且内容与审题良好：可在同档内上浮 0.5 分

## 输出步骤
1. 先给出整体档次判断及理由。
2. 再给出各维度得分（必须与档次一致）。
3. 最后给出总分（各维度分之和，允许 0.5 分微调）。

## 输出格式
{
  "totalScore": 13.5,
  "scoreTier": "1st",
  "tierLabel": "第一档（优）",
  "dimensionScores": {
    "topicAdherence": 1.5,
    "content": 4.5,
    "language": 3.5,
    "structure": 2.0,
    "presentation": 1.5
  },
  "suggestions": [
    { "priority": "high", "category": "内容", "suggestion": "补充遗漏的第二个要点，并给出具体例子" }
  ],
  "comment": "整体评语"
}
```

### 3.2 评分计算公式

```typescript
function calculateFinalScore(
  topicAdherenceScore: number, // 0-2.0
  contentScore: number,        // 0-5.0
  languageScore: number,       // 0-4.0
  structureScore: number,      // 0-2.5
  presentationScore: number,   // 0-1.5
  handwritingScore: number | null, // OCR书写质量 0-1.5 (null=typed)
  errorCount: number,
  missingPoints: number,
  hasFormatError: boolean,
  wordCount: number,
): number {
  // 基础分 = 各维度得分之和
  // 如果是OCR提交，卷面分 = AI书写质量评估分；否则默认满分
  const effectivePresentation = handwritingScore ?? presentationScore;
  let baseScore = topicAdherenceScore + contentScore + languageScore + structureScore + effectivePresentation;
  
  // 扣分项
  const deduction = 
    errorCount * 0.5 +           // 语法错误扣分
    missingPoints * 1.0 +        // 要点遗漏扣分
    (hasFormatError ? 1.0 : 0);  // 格式错误扣分
  
  let finalScore = baseScore - deduction;
  
  // 词数不足80，降一档（约-3分）
  if (wordCount < 80) {
    finalScore = Math.min(finalScore, 9.5); // 最多第三档
  }
  
  // 词数在100-125最佳区间，且审题扣题与内容质量均较好，可加0.5分
  if (wordCount >= 100 && wordCount <= 125 && topicAdherenceScore >= 2.5 && contentScore >= 1.0) {
    finalScore = Math.min(15, finalScore + 0.5);
  }
  
  // 确保分数在0-15范围内
  return Math.max(0, Math.min(15, Math.round(finalScore * 2) / 2));
}
```

```typescript
// packages/ai/router.ts
class AIProviderRouter {
  private providers: AIProvider[] = [];
  private fallbackChain: AIProvider[][] = [];
  
  /**
   * 路由策略:
   * 1. 默认: DeepSeek-V3 (性价比高)
   * 2. 高精度场景: GPT-4o (内容分析、综合评分)
   * 3. 备用: Claude 3.5 Sonnet
   * 4. 国内备用: 通义千问
   */
  async route(correctionType: 'content' | 'topicAdherence' | 'language' | 'structure' | 'scorer'): Promise<AIProvider> {
    switch (correctionType) {
      case 'content':
      case 'topicAdherence':
        return this.getProvider('openai');   // 内容/审题扣题分析用GPT-4o
      case 'language':
        return this.getProvider('deepseek');  // 语言分析用DeepSeek
      case 'structure':
        return this.getProvider('deepseek');  // 结构分析用DeepSeek
      case 'scorer':
        return this.getProvider('openai');    // 综合评分用GPT-4o
    }
  }
  
  async executeWithFallback<T>(
    correctionType: string,
    fn: (provider: AIProvider) => Promise<T>,
  ): Promise<T> {
    const primaryProvider = await this.route(correctionType);
    try {
      return await fn(primaryProvider);
    } catch (error) {
      logger.warn(`Primary provider failed, trying fallback`, { error });
      const fallbackProvider = this.getFallback();
      return await fn(fallbackProvider);
    }
  }
}
```

## 5. 错误处理与降级

### 5.1 降级策略

```
Level 1: 正常模式
├── 三个维度并行批改
├── 综合评分
└── 完整报告生成

Level 2: 降级模式（主模型超时）
├── 切换到备用模型
└── 功能不变

Level 3: 部分降级（全部模型不可用）
├── 仅返回语言维度分析（基于规则引擎）
├── 拼写检查（本地词典）
└── 不返回综合评分

Level 4: 最低降级
├── 仅返回词数统计
```

### 6.3 书写质量计算

```typescript
function calculateHandwritingScore(details: HandwritingDetails): number {
  const { neatnessScore, letterFormScore, alignmentScore, correctionMarks } = details;
  
  // 三个维度取平均
  let score = (neatnessScore + letterFormScore + alignmentScore) / 3;
  
  // 涂改扣分：超过3处每处扣0.1分
  if (correctionMarks > 3) {
    score -= (correctionMarks - 3) * 0.1;
  }
  
  // 最终得分范围 0-1.5
  return Math.max(0, Math.min(1.5, Math.round(score * 10) / 10));
}
```
